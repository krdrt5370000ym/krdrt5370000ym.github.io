<script>
const getTrackDetails = async (songId) => {
  const url = `https://shazam.p.rapidapi.com/shazam-songs/get-details?id=${songId}&locale=pl-PL`;
  const options = {
    method: 'GET',
    headers: {
      'x-rapidapi-key': '688268f4d0msh69176cddbc5ed7bp1008d5jsn1f63b7d6afe9', // Pamiętaj o bezpieczeństwie!
      'x-rapidapi-host': 'shazam.p.rapidapi.com'
    }
  };

  try {
    const response = await fetch(url, options);
    if (!response.ok) throw new Error(`Błąd: ${response.status}`);
    
    const result = await response.json();
    console.log(result);
    return result; // Zwracamy dane, by użyć ich w innej części aplikacji
  } catch (error) {
    console.error('Błąd pobierania danych:', error);
  }
};

// Przykładowe wywołanie (id musi być ciągiem znaków lub liczbą)
